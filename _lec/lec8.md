```racket
(define (phi p m n)
  (cond
    [(zero? p) (+ m n)]
    [(and (zero? n) (zero? (sub1 p))) 0]
    [(and (zero? n) (zero? (sub1 (sub1 p)))) 1]
    [(zero? n) m]
    [else (phi (sub1 p) m (phi p m (sub1 n)))]))
```
